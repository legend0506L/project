    <button onclick="ans()">answer</button>
    <script>
        function ans() {
            var num1 = document.getElementById('txt').value;
            var num = parseInt(num1);
            var sum = 0,i;
            for (i = 2; i <= num; i++) {
                if (i % 2 != 0) {
                    sum = sum + i;
                }
            }
            document.getElementById('p').innerHTML = "sum of odd no "+sum;
        }
    </script>
