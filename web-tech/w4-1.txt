    <script>
        function armstrong() {
            var flag, rem, add = 0, no;
            no = document.getElementById("n").value;
            flag = no;
            while (no > 0) {
                rem = no % 10;
                add = add + rem * rem * rem;
                no = parseInt(no / 10);
            }
            if (add == flag) {
                var p = document.getElementById("s").innerHTML = "it is  an armstrong";
                // alert("it is an armstrong")
            }
            else {
                var p = document.getElementById("s").innerText = "it is not an armstrong";
            }
        }
    </script>
